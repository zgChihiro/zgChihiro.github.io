UE4中使用正则表达式

2018-09-27 1 min read # UE4 


bool AMyActor::Regular(const FString str, const FString Reg)
{
	FRegexPattern Pattern(Reg);
	FRegexMatcher regMatcher(Pattern, str);
	regMatcher.SetLimits(0,str.Len());
	return regMatcher.FindNext();
}
