UE4之使用C++创建定时器

2018-09-07 1 min read # UE4

蓝图中我们延迟执行一个函数可以直接用Delay
在C++中我们也可以用定时器来操作

	//.h
	FTimerHandle MyTimerHand;		//声明一个定时器句柄
	void TempFun();	

//.cpp
void ATimerTest::BeginPlay()
{
	Super::BeginPlay();

	GetWorldTimerManager().SetTimer(MyTimerHand,this,&ATimerTest::TempFun,1.0f,true);
	
}

int i = 0;
void ATimerTest::TempFun()
{
	
	i=i+1;

	GEngine->AddOnScreenDebugMessage(-1, 10.0f, FColor::Green, FString::FromInt(i));
	if (i>9)
	{
		GetWorldTimerManager().ClearTimer(MyTimerHand);
		GEngine->AddOnScreenDebugMessage(-1, 5.0f, FColor::Green, TEXT("HelloWorld"));
	}

}
