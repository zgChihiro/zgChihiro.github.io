UE4读写配置

2018-06-03 1 min read # UE4 
//写配置项
void AFirstConfig::WriteConfig(int i)
{
	GConfig->SetString(TEXT("GUN_NAME"), TEXT("m4a1"), TEXT("1~5"), FPaths::GameDir() / "FirstConfig.ini");//区块、名字、值、路径
	GConfig->Flush(true,FPaths::GameDir() / "FirstConfig.ini");			//立即生效
}
//都配置项
FString AFirstConfig::ReadConfig()
{
	FString Result;		//返回信息
	GConfig->GetString(TEXT("GUN_NAME"), TEXT("m4a1"), Result, FPaths::GameDir() / "FirstConfig.ini");//区块、名字、值、路径
	return Result;
}
